function [FMx1,FMx2,FMy1,FMy2,Cx,Binx,Cy,Biny,Vec_errores,Mejor_Crom,Min_error] = Cal_Crom_Eval(N_Crom,Cant_Incog,Cant_Var,Cromosomas,x_FM_x,x_FM_y,x,y,Cmin_x,Cmax_x,Cmin_y,Cmax_y,Tam_Crom)
Cx=[];
Binx=[];
Cy=[];
Biny=[];
Vec_errores=[];
pos=0;
for n=1:N_Crom
for i=1:Cant_Incog/Cant_Var
b=BintoDec(Cromosomas(n,(i*(Tam_Crom)-(Tam_Crom-1)):Tam_Crom*i));
Binx(n,i)=b;
Cx(n,i)=Cmin_x+(b/((2^Tam_Crom)-1))*(Cmax_x-Cmin_x);
b=BintoDec(Cromosomas(n,((i+Cant_Var)*(Tam_Crom)-(Tam_Crom-1)):Tam_Crom*(i+Cant_Var)));
Biny(n,i)=b;
Cy(n,i)=Cmin_y+(b/((2^Tam_Crom)-1))*(Cmax_y-Cmin_y);
end
Min_act=0;
FMx1 = trimf(x_FM_x,[Cmin_x Cmin_x Cx(n,1)]);
FMx2 = trimf(x_FM_x,[Cmax_x-Cx(n,2) Cmax_x Cmax_x]);
FMy1 = trimf(x_FM_y,[Cmin_y Cmin_y Cy(n,1)]);
FMy2 = trimf(x_FM_y,[Cmax_y-Cy(n,2) Cmax_y Cmax_y]);
[Filas,Columnas]=size(x_FM_x);
Cercano1=abs(x_FM_x(1)-x(2));
Cercano2=abs(x_FM_x(1)-x(3));
pos1=1;
pos2=1;
for j=2:Columnas
if(abs(x_FM_x(j)-x(2))<Cercano1)
Cercano1=abs(x_FM_x(j)-x(2));
pos1=j;
end
if(abs(x_FM_x(j)-x(2))<Cercano2)
Cercano2=abs(x_FM_x(j)-x(3));
pos2=j;
end
end
Membresia1=FMx1(pos1);
Membresia2=FMx2(pos2);
Cercano1=abs(FMy1(1)-Membresia1);
Cercano2=abs(FMy2(1)-Membresia2);
pos1=1;
pos2=1;
[Filas,Columnas]=size(FMy1);
for j=2:Columnas
if(abs(FMy1(j)-Membresia1)<Cercano1)
Cercano1=abs(x_FM_y(j)-Membresia1);
pos1=j;
end
if(abs(FMy2(j)-Membresia2)<Cercano2)
Cercano2=abs(x_FM_y(j)-Membresia2);
pos2=j;
end
end
Correspondencia1=x_FM_y(pos1);
Correspondencia2=x_FM_y(pos2);
Min_act=Min_act+(Correspondencia1-y(2))^2;
Min_act=Min_act+(Correspondencia2-y(2))^2;
[Filas,Columnas]=size(x_FM_y);
Cercano1=abs(x_FM_y(1)-y(2));
Cercano2=abs(x_FM_x(1)-y(3));
pos1=1;
pos2=1;
for j=2:Columnas
if(abs(x_FM_y(j)-y(2))<Cercano1)
Cercano1=abs(x_FM_y(j)-y(2));
pos1=j;
end
if(abs(x_FM_y(j)-y(2))<Cercano2)
Cercano2=abs(x_FM_y(j)-y(3));
pos2=j;
end
end
Membresia1=FMy1(pos1);
Membresia2=FMy2(pos2);
Cercano1=abs(FMx1(1)-Membresia1);
Cercano2=abs(FMx2(1)-Membresia2);
pos1=1;
pos2=1;
[Filas,Columnas]=size(FMx1);
for j=2:Columnas
if(abs(FMx1(j)-Membresia1)<Cercano1)
Cercano1=abs(x_FM_x(j)-Membresia1);
pos1=j;
end
if(abs(FMx2(j)-Membresia2)<Cercano2)
Cercano2=abs(x_FM_x(j)-Membresia2);
pos2=j;
end
end
Correspondencia1=x_FM_x(pos1);
Correspondencia2=x_FM_x(pos2);
Min_act=Min_act+(Correspondencia1-x(2))^2;
Min_act=Min_act+(Correspondencia2-x(2))^2;
Vec_errores(n)=Min_act;
if(n==1)
Min_error=Min_act;
Mejor_Crom=n;
else
if(Min_act<Min_error)
Min_error=Min_act;
Mejor_Crom=n;
end
end
end
end
Funcion para calcular la posibilidad de seleccionar un cromosoma para la combinacion
function [Prob] = Prob(Vec_errores,N_Crom)
Cal=0;
Prob=[];
for i=1:N_Crom
Cal=Vec_errores(i)/sum(Vec_errores);
Prob(i)=Cal;
end
end
Funcion para transformar un numero binario a uno decimal
function [dec] = BintoDec(Cadena)
dec=0;
Cal=0;
[Filas,Columnas]=size(Cadena);
for i=0:Columnas-1
if(Cadena(1,Columnas-i)==0)
Cal=0;
else
Cal=2^(i);
dec=dec+Cal;
end
end
end
Funcion para mezclar los Cromosomas en una nueva generacion
function [Cromosomas] = Comb(N_Crom,Probabilidades,Cromosomas,Cadena_Total)
Nuevo_Cromosomas=[];
Corte=round(1+rand()*(Cadena_Total-1));
pos1=1;
pos2=1;
for i=1:N_Crom
Probabilidades(i)=Probabilidades(i)*i;
end
for i=1:N_Crom
Val1=0+rand()*(1-0);
Val2=0+rand()*(1-0);
for j=1:N_Crom-1
if(Val1>0 & Val1<=Probabilidades(1))
Pos1=j;
end
if(Val1>Probabilidades(j) & Val1<=Probabilidades(j+1))
Pos1=j;
end
if(Val2>0 & Val2<=Probabilidades(1))
Pos2=j;
end
if(Val2>Probabilidades(j) & Val2<=Probabilidades(j+1))
Pos2=j;
end
end
end
for n=1:Cadena_Total
if(n==Corte)
Nuevo_Cromosomas(i,n)=Cromosomas(pos2,n);
else
Nuevo_Cromosomas(i,n)=Cromosomas(pos1,n);
end
end
Cromosomas=Nuevo_Cromosomas;
end
