la función recibe ambos parametros y despues pide indicar que tipo de complemento se desea
function [YC] = Problema4(x,A)
TC=input("elija el tipo de complemento deseado 2=sugeno 3=estandar 1=yager");
y=A;
if TC==1
w=input("seleccione un valor para w mayor a 0");
C=(1-(y.^w)).^(1/w);
YC=C;
end
if TC==2
s=input("indique un valor de s mayor a -1");
C=(1-y)./(1+(s*y));
YC=C;
end
if TC==3
C=1-y;
YC=C;
end
subplot(2,1,1)
plot(x,y,"g");
title("conjunto original");
subplot(2,1,2);
plot(x,YC,"r");
title("complemento del conjunto")
end

Función Problema 5

La función encargada de realizar los calculos de la intersección es:
function [YC] = Problema5(x,A,B)
TC=input("elija el tipo de intersección deseado 1=min 2=Producto algebraico 3=Producto acotado");
if TC==1
for i=1:length(A)
if A(i)>=B(i)
YC(i)=B(i);
else
YC(i)=A(i);
end
end
end
if TC==2
YC=A.*B;
end
if TC==3
y=A+B-1;
for i=1:length(A)
if y(i)>=0
YC(i)=y(i);
else
YC(i)=0;
end
end
end
subplot(3,1,1)
plot(x,A,"g");
title("conjunto original A");
subplot(3,1,2);
plot(x,B,"r");
title("conjunto original B")
subplot(3,1,3);
plot(x,YC,"b");
title("Intersección de los conjuntos")
end

ylabel("rango de pertenencia");

end
